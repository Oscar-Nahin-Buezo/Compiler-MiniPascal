program SumaDosNumeros;

function Sumar(num1, num2: integer;): integer;
var
  resultado: integer;
begin
  resultado := num1 + num2 * 5/4;
  Sumar := resultado;
end;

var
  a, b, suma: integer;

begin
  writeln('Ingrese el primer número: ');
  readln(a);

  writeln('Ingrese el segundo número: ');
  readln(b);

  suma := Sumar(a, b);

  writeln('La suma de ', a, ' y ', b, ' es: ', suma);

end.
